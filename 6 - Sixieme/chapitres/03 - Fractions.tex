\partie{Fractions et quotients}

\definition{Le quotient de $a$ par $b$ est le nombre qui, multiplié par $b$ est égal à $a$.}

\notation{Il est noté $\dfrac{a}{b}$.}

\exemple{$\dfrac{3}{4} \times 4 = \dfrac{3}{\cancel{4}} \times \cancel{4} = 3$}

\propriete{Multiplier le numérateur et le dénominateur par le \emph{même} nombre, donne une fraction égale à celle d'origine.}

\exemple{$\dfrac{4}{5} = \dfrac{12}{15}$ (on a multiplié par 3 en haut et en bas)}

\propriete{Si le numérateur est plus petit que le dénominateur, alors la fraction est plus petite que 1.}

\propriete{Si le numérateur est plus grand que le dénominateur, alors la fraction est plus grande que 1.}

\begin{listedexemples}
    \item $\dfrac{9}{10} \Rightarrow 9 < 10 $ donc $\dfrac{9}{10} < 1$
    \item $\dfrac{12}{11} \Rightarrow 12 > 11 $ donc $\dfrac{12}{11} > 1$
\end{listedexemples}
