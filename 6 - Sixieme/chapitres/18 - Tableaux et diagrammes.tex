\chapitre{Tableaux et diagrammes}

\partie{Tableaux}

\definition{Un tableau est un outil qui permet de classifier des données et d'en faciliter la lecture.}

\souspartie{Tableau en lignes}

\exemple{
    \begin{center}
        \begin{tabularx}{0.5\linewidth}{|X|X|} \hline
            Température (\unit{\degreeCelsius}) & mois \\\hline
            10 & janvier \\\hline
            15 & février \\\hline 
            17 & mars \\\hline
        \end{tabularx}
    \end{center}
    Au mois de janvier, la température moyenne s'élève à \qty{10}{\degreeCelsius}.
}

\souspartie{Tableau en colonnes}

\exemple{
    \begin{center}
        \begin{tabularx}{0.7\linewidth}{|l|X|X|X|} \hline
            salle & 24 & 15 & 17bis \\\hline 
            nombre de chaises & 30 & 31 & 12 \\\hline
        \end{tabularx}
    \end{center}
    En salle 17bis, il y a 12 chaises.
}

\souspartie{Tableau à double entrée}

\exemple{
    \begin{center}
        \begin{tabularx}{0.7\linewidth}{|X|X|X|X|}\hline
             & En ville & Autoroute \\\hline 
             France & \qty{50}{\kilo\metre\per\hour} & \qty{130}{\kilo\metre\per\hour} \\\hline 
             Suisse & \qty{50}{\kilo\metre\per\hour} & \qty{120}{\kilo\metre\per\hour} \\\hline 
             Royaume-Uni & \qty{48}{\kilo\metre\per\hour} & \qty{113}{\kilo\metre\per\hour} \\\hline
        \end{tabularx}
    \end{center}
    En ville, au Royaume-Uni, la vitesse limite est de \qty{48}{\kilo\metre\per\hour}.
}

\partie{Graphiques et diagrammes}

\souspartie{Graphique cartésien}
\definition{Un graphique cartésien est constitué d'un repère orthogonal sur lequel on place plusieurs points que l'on va relier pour former une courbe.}

\illustration{
    \begin{center}
        \begin{tabularx}{0.9\linewidth}{|l|X|X|X|X|X|}\hline
            Jour & vendredi 11 avril & samedi 15 avril & dimanche 16 avril & lundi 17 avril & mardi 18 avril \\\hline
            Température (en \unit{\degreeCelsius}) & 11 & 15 & 16 & 17 & 17 \\\hline
        \end{tabularx}
        %
        \begin{tikzpicture}
            \draw[-Latex] (-1,0) -- (12,0);
            \draw[-Latex] (0,-1) -- (0,6.5);
            \foreach \y in {1,...,18} {
                \draw (0.1,{\y/3}) -- +(-0.2,0);
            }
            \foreach \x in {1,...,5} {
                \draw ({2*\x},-0.1) -- +(0,0.2);
            }
        \end{tikzpicture}
    \end{center}
}

\clearpage
\souspartie{Diagrammes en bâtons}

\definition{Un diagramme en bâtons (ou en barres) permet de représenter les données d'un tableau par des barres de hauteurs proportionnelles aux données.}

\exemple{\color{noir}
\begin{center}%
Nombre d'habitants dans 3 pays en 2020\\
\begin{tabular}{|c|c|}\hline
    \textsc{Pays} & \textsc{Nombre d'habitants} (en millions) \\\hline
    France & 67 \\\hline
    Pologne & 38 \\\hline
    Corée du sud & 51 \\\hline
\end{tabular}%
\end{center}%
\begin{flushright}
    \tiny{Sources : Instituts nationaux officiels de statistiques}
\end{flushright}
\begin{center}
    \begin{tikzpicture}
        \draw[->] (-1,0) -- (11,0);
        \draw[->] (0,-1) -- (0,8);
        \foreach \y in {10,20,...,70} {
            \draw (-0.2,{0.1*\y}) -- (0.2,{0.1*\y});
            \node[left] at (-0.2,{0.1*\y}) {\y};
        }
        \node[below] at (2,0) {France};
        \node[below] at (5,0) {Pologne};
        \node[below] at (8,0) {Corée du Sud};
        \draw[fill=noir] (1.8,0) rectangle (2.2,6.7);
        \draw[fill=noir] (4.8,0) rectangle (5.2,3.8);
        \draw[fill=noir] (7.8,0) rectangle (8.2,5.1);
        \draw[dashed] (0,6.7) -- (1.8,6.7);
        \draw[dashed] (0,3.8) -- (4.8,3.8);
        \draw[dashed] (0,5.1) -- (7.8,5.1);
        \node[thin, above left] at (0,5.1) {\small{51}};
        \node[thin, left] at (0,6.7) {\small{67}};
        \node[thin, below left] at (0,3.8) {\small{38}};
    \end{tikzpicture}
\end{center}
}

\clearpage
\souspartie{Diagrammes circulaires}

\definition{Un diagramme circulaire permet de représenter les données d'un tableau par des secteurs de disque d'angles proportionnels aux données du tableau.}

\exemple{}

\begin{center}
    Estimation du nombre d'habitants par continent en 2021\\
    \begin{tabular}{|c|c||c|}\hline
        \textsc{Continent} & \makecell{\textsc{Nombre d'habitants}\\ (en millions)} & \textsc{Angle} (en °) \\\hline
        Asie & 4 679 & 213 \\\hline
        Afrique & 1 373 & 62 \\\hline
        Europe & 747 & 34 \\\hline
        Amérique latine et Caraïbes & 659 & 30 \\\hline
        États-Unis et Canada & 371 & 16 \\\hline
        Océanie & 43 & 1 \\\hline 
        Antartique & 0 & 0 \\\hline
        \textsc{TOTAL} & 7872 & 360 \\\hline 
    \end{tabular}
\end{center}

\begin{flushright}
    \tiny{Source : https://www.un.org/development/desa/publications/world-population-prospects-the-2017-revision.html}
\end{flushright}

\newcommand{\macroLegende}[3]{ % position, nom, couleur
    \draw[fill=#3] (4,#1) rectangle (4,{#1+0.3});
    \node[above right] (4,#1) {#2};
}

\begin{center}
    \begin{tikzpicture}
        \draw (0,0) circle (3);
        \foreach \y in {-1.5,-1,...,1.5} {
            \draw (4.5,\y) -- (8,\y);
        }

        \draw (4,1) rectangle (4.3,1.3);

        \draw[fill=noir] (0,0) circle (0.03);
        \draw[very thin] (0,0) -- (3,0);
    \end{tikzpicture}
\end{center}
