\newcommand{\questionA}{\directlua{
    local a = math.random(1,10)
    local b = math.random(1,10)
    local operateur = {"+", "-", "\\times", "/"}

    local op = math.random(1,3)
    local resultat = 0
    
    if op == 1 then
        resultat = a+b
    elseif op == 2 then
        resultat = a-b
    elseif op == 3 then
        resultat = a*b
    end

    tex.sprint("\\VMQCMAjouterQuestion{$", a, operateur[op], b, "$}{u}{v}{w}{", resultat, "}{D}{1}")
}}
